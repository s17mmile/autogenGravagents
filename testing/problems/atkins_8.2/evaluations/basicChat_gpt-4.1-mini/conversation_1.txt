\kappa L_1 & =\left\{\frac{2 m_{\mathrm{e}}(V-E)}{\hbar^2}\right\}^{1 / 2} L_1 \\

\end{aligned}
$$

$$
\text { current at } \begin{aligned}

& =\mathrm{e}^{-2 \times\left(7.25 \times 10^{-9} \mathrm{~m}^{-1}\right) \times\left(1.0 \times 10^{-10} \mathrm{~m}\right)}=0.23
\end{aligned}
$$
